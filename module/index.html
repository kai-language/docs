    Import and Modules

    Import a File

    Import from Git

    // github
#import git("author/repo")

// external
#import git("https://external.com/author/repo")

    Alias an Import

    #import "math" as mathy
print(mathy.pi)

    Import a C Header

    #import "stdio.h"
